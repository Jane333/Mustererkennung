\begin{verbatim}
ueberlebt_maennlich = ueberlebt(ueberlebt(:,3)==0,:)

P_m_U = size(ueberlebt_maennlich, 1) / size(ueberlebt, 1)
% P(m|U) =  0.32069
\end{verbatim}
